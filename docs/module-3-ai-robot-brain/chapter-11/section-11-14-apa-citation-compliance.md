## APA Citation Standards Review

### In-Text Citation Format

According to APA 7th edition, in-text citations should follow the author-date format:

- **Single Author**: (Author, Year) or Author (Year)
- **Two Authors**: (Author & Author, Year) or Author and Author (Year)
- **Three or More Authors**: (Author et al., Year) or Author et al. (Year)
- **Direct Quotation**: Include page number (Author, Year, p. #) or (Author, Year, pp. #-#)

### Reference List Format

References should be formatted as:

- **Journal Articles**: Author, A. A. (Year). Title of article. *Title of Periodical*, Volume(Issue), pages. DOI or URL
- **Books**: Author, A. A. (Year). *Title of work*. Publisher.
- **Web Pages**: Author, A. A. (Year, Month Day). Title of webpage. Site Name. URL
- **Technical Documentation**: Author/Organization. (Year). Title of documentation. URL

# citation_validator.py
import re
from typing import List, Dict

class CitationValidator:
    def __init__(self):
        self.in_text_pattern = r'\([A-Za-z0-9\-,&\s\.]+\s*,\s*\d{4}\)'
        self.reference_patterns = {
            'journal': r'^[A-Z][a-z]+,\s*[A-Z]\.[A-Z]*\.,\s*&\s*[A-Z][a-z]+,\s*[A-Z]\.[A-Z]*\.\s*\(\d{4}\)\.',
            'web': r'^[A-Z][a-z]+,\s*[A-Z]\.\s*\(\d{4}\)\.\s*[A-Z][a-z\s\d\-\(\)]+\.\s*[A-Z][a-z\s]+\.\s*https?://'
        }
    
    def validate_in_text_citations(self, text: str) -> List[str]:
        """Validate in-text citations in the provided text"""
        citations = re.findall(self.in_text_pattern, text)
        valid_citations = []
        
        for citation in citations:
            # Check if citation follows APA format
            if self._is_valid_in_text_format(citation):
                valid_citations.append(citation)
        
        return valid_citations
    
    def _is_valid_in_text_format(self, citation: str) -> bool:
        """Check if in-text citation follows APA format"""
        # Basic format check: (Author, Year) or (Author & Author, Year)
        if ',' in citation and any(char.isdigit() for char in citation):
            year = re.search(r'\d{4}', citation).group()
            year_int = int(year)
            if 1900 <= year_int <= 2030:  # Reasonable year range
                return True
        return False
    
    def validate_reference_list(self, references: List[str]) -> Dict:
        """Validate the format of reference list entries"""
        results = {
            'valid': [],
            'invalid': [],
            'format_issues': []
        }
        
        for ref in references:
            is_valid, issues = self._validate_reference_format(ref)
            if is_valid:
                results['valid'].append(ref)
            else:
                results['invalid'].append(ref)
                results['format_issues'].extend(issues)
        
        return results
    
    def _validate_reference_format(self, reference: str) -> tuple:
        """Validate individual reference format"""
        issues = []
        
        # Check for required elements
        if not re.search(r'[A-Z][a-z]+,\s*[A-Z]\.', reference):
            issues.append("Missing author in correct format")
        
        if not re.search(r'\(\d{4}\)', reference):
            issues.append("Missing year in parentheses")
        
        if not re.search(r'https?://', reference) and not re.search(r'\.', reference.split()[-1]):
            issues.append("Missing URL or publisher information")
        
        return len(issues) == 0, issues
